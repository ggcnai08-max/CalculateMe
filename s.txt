<script>
  const STATUS = {
    general:      { min:1.2, max:1.5, mode:'mid',   color:'var(--status1)', note:'ค่ากึ่งกลางของช่วง เหมาะสำหรับรักษาสมดุลของร่างกายในระยะเฝ้าระวัง' },
    treatment:    { min:1.5, max:2.0, mode:'upper',  color:'var(--status2)', note:'เน้นค่าบนของช่วง เพื่อรองรับความต้องการที่เพิ่มขึ้นระหว่างการรักษา' },
    malnutrition: { min:1.5, max:2.0, mode:'upper',  color:'var(--status3)', note:'เน้นค่าบนของช่วง เพื่อเร่งฟื้นฟูมวลกล้ามเนื้อและร่างกาย' }
  };
  const SCALE_MAX = 250; // grams, for gauge visualization only

  let currentStatus = 'general';
  const weightInput = document.getElementById('weightInput');
  const weightSlider = document.getElementById('weightSlider');

  function tween(el, to, decimals){
    const from = parseFloat(el.textContent) || 0;
    const dur = 450;
    const start = performance.now();
    function frame(now){
      const p = Math.min(1, (now - start) / dur);
      const eased = 1 - Math.pow(1 - p, 3);
      const val = from + (to - from) * eased;
      el.textContent = val.toFixed(decimals);
      if(p < 1) requestAnimationFrame(frame);
    }
    requestAnimationFrame(frame);
  }

  function clamp(v, lo, hi){ return Math.max(lo, Math.min(hi, v)); }

  function calculate(){
    const w = clamp(parseFloat(weightInput.value) || 0, 0, 400);
    const s = STATUS[currentStatus];
    const min = w * s.min;
    const max = w * s.max;
    const rec = s.mode === 'mid' ? (min + max) / 2 : max;

    tween(document.getElementById('minVal'), min, 1);
    tween(document.getElementById('maxVal'), max, 1);
    tween(document.getElementById('recVal'), rec, 1);
    document.getElementById('recNote').textContent = s.note;

    const fillPct = clamp(rec / SCALE_MAX, 0, 1) * 100;
    const minPct = clamp(min / SCALE_MAX, 0, 1) * 100;
    const maxPct = clamp(max / SCALE_MAX, 0, 1) * 100;
    document.getElementById('bowlFill').style.height = fillPct + '%';
    document.getElementById('markMin').style.bottom = minPct + '%';
    document.getElementById('markMax').style.bottom = maxPct + '%';
    document.getElementById('markMinLabel').textContent = min.toFixed(0);
    document.getElementById('markMaxLabel').textContent = max.toFixed(0);

    document.getElementById('eqEgg').textContent = (rec / 6).toFixed(1);
    document.getElementById('eqChicken').textContent = (rec / 31).toFixed(1);
    document.getElementById('eqTofu').textContent = (rec / 8).toFixed(1);
    document.getElementById('eqMilk').textContent = (rec / 8).toFixed(1);

    document.querySelectorAll('.info-card').forEach(c=>{
      c.classList.toggle('on', c.dataset.status === currentStatus);
    });

    const pct = ((w - weightSlider.min) / (weightSlider.max - weightSlider.min)) * 100;
    weightSlider.style.setProperty('--fill', pct + '%');
  }

  document.querySelectorAll('.seg-btn').forEach(btn=>{
    btn.addEventListener('click', () => {
      document.querySelectorAll('.seg-btn').forEach(b=>b.classList.remove('active'));
      btn.classList.add('active');
      currentStatus = btn.dataset.status;
      calculate();
    });
  });

  weightInput.addEventListener('input', () => {
    let v = clamp(parseFloat(weightInput.value) || 0, 20, 150);
    weightSlider.value = v;
    calculate();
  });
  weightSlider.addEventListener('input', () => {
    weightInput.value = weightSlider.value;
    calculate();
  });

  calculate();
</script>
